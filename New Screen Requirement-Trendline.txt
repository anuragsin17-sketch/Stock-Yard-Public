Part 1: Update Your Backend to Generate the JSON Data
Add this function to your server or background worker file. It executes the geometric trendline algorithm over the Nifty 500 and writes the results to a file named trendline_screen.json.

import json
from geometric_engine import GeometricTrendlineEngine

def update_trendline_json_feed():
    """
    Runs the scanner across your ticker universe and updates the JSON file
    that your HTML front-end screen reads from.
    """
    engine = GeometricTrendlineEngine(buffer_percentage=10.0, critical_trigger_percentage=1.0)
    
    # Replace this array with your system's full list of Nifty 500 tickers
    watchlist = ["BHEL.NS", "CDSL.NS", "SBIN.NS", "AXISBANK.NS", "TATACONSUM.NS", "ICICIBANK.NS"]
    
    screen_data = []
    
    for ticker in watchlist:
        metrics = engine.extract_pattern_metrics(ticker)
        if metrics:
            # Format clean key names that your HTML Javascript can map directly into table cells
            screen_data.append({
                "ticker": metrics["ticker"].replace(".NS", ""), # Clean ticker name for UI
                "currentPrice": metrics["current_price"],
                "triggerPrice": metrics["expected_entry"],
                "distance": metrics["distance_pct"],
                "targetExit": metrics["expected_exit"],
                "status": metrics["status"]
            })
            
            # Hook your existing live production Telegram notification system right here:
            if metrics["status"] == "CRITICAL_TOUCH":
                # send_telegram_alert(f"🚨 {ticker} hit trendline entry target!")
                pass
                
    # Save to your static asset directory or API data path so the HTML UI can load it
    with open("trendline_screen.json", "w") as f:
        json.dump(screen_data, f, indent=4)
        
    print("✅ trendline_screen.json updated successfully.")

# Run it to generate the initial data load file
if __name__ == "__main__":
    update_trendline_json_feed()

    // Fetch the JSON payload and build out your data dashboard screen columns
    async function loadTrendlineScreenData() {
        try {
            // Replace with your true backend data path or API URL endpoint
            const response = await fetch('trendline_screen.json');
            const data = await response.json();
            
            const tableBody = document.getElementById('table-body');
            tableBody.innerHTML = ''; // Wipe loading state placeholder row
            
            // Sort data so the stocks closest to a trendline touch sit at the top of your UI
            data.sort((a, b) => a.distance - b.distance);
            
            data.forEach(stock => {
                const row = document.createElement('tr');
                
                // Assign correct color badge configurations depending on proximity
                const badgeClass = stock.status === 'CRITICAL_TOUCH' ? 'status-critical' : 'status-watchlist';
                const statusLabel = stock.status === 'CRITICAL_TOUCH' ? '🎯 CRITICAL ENTRY' : '👀 WATCHLIST';
                
                row.innerHTML = `
                    <td style="font-weight: bold; color: #ffffff;">${stock.ticker}</td>
                    <td>₹${stock.currentPrice.toFixed(2)}</td>
                    <td style="color: #38bdf8; font-weight: 600;">₹${stock.triggerPrice.toFixed(2)}</td>
                    <td style="color: ${stock.distance <= 2.0 ? '#f43f5e' : '#f8fafc'}">+${stock.distance.toFixed(2)}%</td>
                    <td style="color: #4ade80; font-weight: 600;">₹${stock.targetExit.toFixed(2)}</td>
                    <td><span class="badge ${badgeClass}">${statusLabel}</span></td>
                `;
                
                tableBody.appendChild(row);
            });
            
            document.getElementById('last-updated').innerText = `Last Refreshed: ${new Date().toLocaleTimeString()}`;
            
        } catch (error) {
            console.error('Error loading JSON data source:', error);
            document.getElementById('table-body').innerHTML = `
                <tr><td colspan="6" style="text-align: center; color: #ef4444;">❌ Error fetching data source. Check backend JSON compiler pipeline.</td></tr>
            `;
        }
    }

    // Initial Screen Load
    loadTrendlineScreenData();
    // Auto-refresh the frontend screen layout metrics every 5 minutes to stay accurate
    setInterval(loadTrendlineScreenData, 300000);
</script>
